1 /**
  2  * 淘宝返回基类
  3  * 
  4  * @author fxy
  5  * @class
  6  */
  7 function TaobaoResponse(jsonRsp) {
  8 
  9 	/**
 10 	 * 错误代码
 11 	 * 
 12 	 * @type String
 13 	 */
 14 	this.errorCode = null;
 15 	/**
 16 	 * 错误信息
 17 	 * 
 18 	 * @type String
 19 	 */
 20 	this.msg = null;
 21 	/**
 22 	 * 子系统错误代码
 23 	 * 
 24 	 * @type
 25 	 */
 26 	this.subCode = null;
 27 	/**
 28 	 * 子系统错误信息
 29 	 * 
 30 	 * @type
 31 	 */
 32 	this.subMsg = null;
 33 	/**
 34 	 * 重定向URL
 35 	 * 
 36 	 * @type String
 37 	 */
 38 	this.redirectUrl = null;
 39 	/**
 40 	 * 返回信息
 41 	 * 
 42 	 * @type
 43 	 */
 44 	this.body = {};
 45 	/**
 46 	 * 判断是否调用API成功
 47 	 * 
 48 	 * @function
 49 	 * @return Boolean
 50 	 */
 51 	this.isSuccess = function() {
 52 		return null == this.errorCode;
 53 	}
 54 	/**
 55 	 * 判断是否需要重定向
 56 	 * 
 57 	 * @function
 58 	 * @return Boolean
 59 	 */
 60 	this.isRedirect = function() {
 61 		return null != this.redirectUrl;
 62 	}
 63 	/**
 64 	 * 得到绑定用户需要重定向的URL
 65 	 * 
 66 	 * @param {String}target
 67 	 *            绑定用户成功后重定向的目标地址
 68 	 * @return String
 69 	 */
 70 	this.getRedirectUrl = function(target) {
 71 		return this.redirectUrl + "&sip_redirecturl=" + target;
 72 	}
 73 	/**
 74 	 * 得到绑定用户需要重定向的URL，对绑定后重定向回isv的url做urlEncode编码
 75 	 * 
 76 	 * @param {String}target
 77 	 *            绑定用户成功后重定向的目标地址
 78 	 * @return String
 79 	 */
 80 	this.getEncodeRedirectUrl = function(target) {
 81 		return this.redirectUrl + "&sip_redirecturl=" + encodeURI(target);
 82 	}
 83 	if (jsonRsp != null) {
 84 		if (jsonRsp.hasOwnProperty(TaobaoConstants.ERROR_RSP)) {
 85 			this.errorCode = jsonRsp[TaobaoConstants.ERROR_RSP][TaobaoConstants.ERROR_CODE];
 86 			this.msg = jsonRsp[TaobaoConstants.ERROR_RSP][TaobaoConstants.ERROR_MSG];
 87 			if (jsonRsp[TaobaoConstants.ERROR_RSP]
 88 					.hasOwnProperty(TaobaoConstants.SUB_ERROR_CODE)) {
 89 				this.subCode = jsonRsp[TaobaoConstants.ERROR_RSP][TaobaoConstants.SUB_ERROR_CODE];
 90 				this.subMsg = jsonRsp[TaobaoConstants.ERROR_RSP][TaobaoConstants.SUB_ERROR_MSG];
 91 			}
 92 
 93 		}
 94 		this.body = jsonRsp;
 95 	}
 96 	// this.redirectUrl = response.redirectUrl;
 97 }
