1 /**

  7 function TaobaoJianghuFeedPublishRequest() {
  8 	/**
  9 	 * API
 10 	 * 
 11 	 * @type String
 12 	 */
 13 	this.method = "taobao.jianghu.feed.publish";
 14 	/**
 15 	 * 如果Feed的内容描述的是两人之间发生的关系，则用这个参数指明被参与的一方。注：Feed的发起方不需要指定，默认为当前用户。(须数字，最长20位)
 16 	 * 
 17 	 * @type String
 18 	 */
 19 	this.participator = "";
 20 	/**
 21 	 * 应用自定义feed(动态)的类型(必须为数字取值范围[0,127])
 22 	 * 
 23 	 * @type String
 24 	 */
 25 	this.type = "";
 26 	/**
 27 	 * 标题.feed标题在title里可以使用${actor}和${participator}模板变量,<br>

 29 	 * 如果要使用这个模板变量需要传递participator参数.<br>
 30 	 * 使用示例： <br>
 31 	 * ${actor}踢了${participator} 一下.<br>
 32 	 * 在例子中,动态展示时，系统会将$ {actor}及${participator}替换成相应的人名。<br>
 33 	 * 没有链接.最长200个字符,一个中文字符和一个英文字符的长度都是1.
 34 	 * 
 35 	 * @type String
 36 	 */
 37 	this.title = "";
 38 	/**
 39 	 * 标题链接,最长200个字符,一个中文字符和一个英文字符的长度都是1.
 40 	 * 
 41 	 * @type String
 42 	 */
 43 	this.title_link = "";
 44 	/**
 45 	 * 动态显示的内容数据。在body里可以使用${actor}和${participator}模板变量，其中${actor}表示产生Feed的人，${participator}
 46 	 * 表示动态的参与者，如果要使用这个模板变量需要传递participator参数。使用示例：${actor}踢了${participator}一下。
 47 	 * 在例子中，动态展示时，系统会将${actor}及${participator}替换成相应的人名，并加上对应主页的连接地址。动态显示的内容数据。只支持两个html元素<a>和<img>，其他的html元素将被过滤掉。
 48 	 * script、style、head、select标签的内容被去掉。其他的标签保留内容如：<script>alert</script>和红色
 49 	 * 结果为：红色 （最长2000个字符，一个中文字符和一个英文字符的长度都是1）
 50 	 * 
 51 	 * @type String
 52 	 */
 53 	this.body = "";
 54 	/**
 55 	 * 媒体对象的数据,媒体对象的数据,用json格式表示.<br>
 56 	 * mediaType:0:图片，1:音频，2:flash .<br>
 57 	 * 其中media是媒体对象的图片地址，mediaLink是点击媒体对象时候的目标URL .<br> [
 58 	 * {"mediaName":"baobei","media":"media src","mediaLink":"mdedia
 59 	 * link","mediaDesc":"","mediaType":"0" } ,
 60 	 * {"mediaName":"baobei","media":"media src","mediaLink":"mdedia
 61 	 * link","mediaDesc":"","mediaType":"0"} ]
 62 	 * 
 63 	 * @type String
 64 	 */
 65 	this.medias = "";
 66 	/**
 67 	 * 动态在旺旺上的显示类型.<br>
 68 	 * 如果多个则用逗号分隔.0:主窗口Feed,2:联系人头像前的Feed.<br>
 69 	 * 当wwFeedTypes不为空，title,titleLink 不能为空。
 70 	 * 
 71 	 * @type String
 72 	 */
 73 	this.ww_feed_types = "";
 74 }
