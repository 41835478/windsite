  7 var TaobaoUtils = {
  8 	/**
  9 	 * 生成淘宝sign
 10 	 * 
 11 	 * @param {Array}keys
 12 	 *            属性数组
 13 	 * @param {Object}properties
 14 	 *            属性集合
 15 	 * @param {String}secret
 16 	 *            密钥
 17 	 * @return {String}
 18 	 */
 19 	signature : function(keys, properties, secret) {
 20 		var orgin = secret;
 21 		if (keys != null && properties != null) {
 22 			for (var key in keys) {
 23 				var property = keys[key];
 24 				orgin += property + properties[property];
 25 			}
 26 		} else {
 27 			return null;
 28 		}
 29 		orgin += secret;
 30 		try {
 31 			return hex_md5(orgin).toUpperCase();
 32 		} catch (e) {
 33 			throw new Error("sign error:" + e);
 34 		}
 35 		return null;
 36 	},
 37 	/**
 38 	 * Json对象转换为字符串
 39 	 * 
 40 	 * @param {Json}O
 41 	 * @return {String}
 42 	 */
 43 	json2str : function(o) {
 44 		var r = [];
 45 		if (typeof o == "string")
 46 			return "\""
 47 					+ o.replace(/([\'\"\\])/g, "\\$1").replace(/(\n)/g, " \\n")
 48 							.replace(/(\r)/g, "\\r").replace(/(\t)/g, "\\t")
 49 					+ "\"";
 50 		if (typeof o == "object") {
 51 			if (o == null) {
 52 				r = r.join() + "null";// 因为null只会出现在值,属性不可能为null,所以直接加null
 53 			} else {
 54 				if (!o.sort) {
 55 					for (var i in o) {
 56 						r.push(i + ":" + TaobaoUtils.json2str(o[i]));
 57 					}
 58 					var oString = o.toString;
 59 					if (!!document.all
 60 							&& !/^\n?function\s*toString\(\)\s*\{\n?\s*\[native code\]\n?\s*\}\n?\s*$/
 61 									.test(oString)) {
 62 						r.push("toString:" + oString.toString());
 63 					}
 64 					r = "{" + r.join() + "}"
 65 				} else {
 66 					var len = o.length;
 67 					var i = 0;
 68 					while (i < len) {
 69 						r.push(TaobaoUtils.json2str(o[i]));
 70 						i++;
 71 					}
 72 					r = "[" + r.join() + "]"
 73 				}
 74 			}
 75 			try {
 76 				return r;
 77 			} finally {
 78 				r = null;
 79 				delete r;
 80 			}
 81 		}
 82 		try {
 83 			return o.toString();
 84 		} finally {
 85 			o = null;
 86 			delete o;
 87 		}
 88 	}
 89 
 90 }
