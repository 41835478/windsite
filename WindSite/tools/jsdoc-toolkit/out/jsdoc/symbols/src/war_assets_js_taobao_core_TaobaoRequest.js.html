1 /**
  2  * 淘宝请求基类
  3  * 
  4  * @author fxy
  5  * @class
  6  */
  7 function TaobaoRequest() {
  8 
  9 	/**
 10 	 * APP_KEY
 11 	 * 
 12 	 * @type String
 13 	 */
 14 	this.app_key = TaobaoConstants.API_KEY;
 15 	/**
 16 	 * 方法名
 17 	 * 
 18 	 * @type String
 19 	 */
 20 	this.method = "";
 21 
 22 	/**
 23 	 * 时间戳
 24 	 * 
 25 	 * @type String
 26 	 */
 27 	this.timestamp = "";
 28 	/**
 29 	 * 版本号
 30 	 * 
 31 	 * @type String
 32 	 */
 33 
 34 	this.v = "2.0";
 35 	/**
 36 	 * 返回格式
 37 	 * 
 38 	 * @type
 39 	 */
 40 	this.format = "json";
 41 	/**
 42 	 * 签名
 43 	 * 
 44 	 * @type String
 45 	 */
 46 	this.sign = "";
 47 	/**
 48 	 * 会话
 49 	 * 
 50 	 * @type String
 51 	 */
 52 
 53 	this.session = "";
 54 	/**
 55 	 * 密钥
 56 	 * 
 57 	 * @type String
 58 	 */
 59 
 60 	this.sign_method = "md5";
 61 	/**
 62 	 * 是否是沙箱请求
 63 	 * 
 64 	 * @type Boolean
 65 	 */
 66 
 67 	this.isSandbox = false;
 68 	/**
 69 	 * 参数校验
 70 	 * 
 71 	 * @function
 72 	 */
 73 	this.validate = function() {
 74 		// 参数校验
 75 	}
 76 	/**
 77 	 * 获取带有Sign的有效参数
 78 	 * 
 79 	 * @return Object
 80 	 */
 81 	this.getSignParameters = function() {
 82 		var result = new Object();
 83 		this.timestamp = new Date().format("yyyy-mm-dd hh:mm:ss");
 84 		var keys = [];// 参数名集合
 85 		var i = 0;
 86 		for (var key in this) {
 87 			if (!(this[key] instanceof Function) && this[key] != undefined
 88 					&& this[key] != null && this[key] != "") {
 89 				result[key] = this[key];
 90 				keys[i] = key;
 91 				i++;
 92 			}
 93 		}
 94 		keys.sort();// 排序
 95 		result.sign = TaobaoUtils.signature(keys, result,
 96 				TaobaoConstants.SECRET);
 97 		return result;
 98 	}
 99 	/**
100 	 * 返回没有Sign的有效参数集合
101 	 * 
102 	 * @return {}
103 	 */
104 	this.getParameters = function() {
105 		var result = new Object();
106 		this.timestamp = new Date().format("yyyy-mm-dd hh:mm:ss");
107 		for (var key in this) {
108 			if (!(this[key] instanceof Function) && this[key] != undefined
109 					&& this[key] != null && this[key] != "") {
110 				result[key] = this[key];
111 			}
112 		}
113 		return result;
114 	}
115 }
